// ============================================================ // CPHS — Google Apps Script Backend // Pega este código en Apps Script y despliega como Web App // ============================================================ const SHEET_ID = 'REEMPLAZA_CON_TU_SPREADSHEET_ID'; // ← ver instrucciones // ------------------------------------------------------------ // CORS + Router principal // ------------------------------------------------------------ function doPost(e) { const data = JSON.parse(e.postData.contents); let result; switch (data.action) { case 'guardarReunion': result = guardarReunion(data.payload); break; case 'cargarPendientes': result = cargarPendientes(); break; default: result = { ok: false, msg: 'Acción desconocida' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const action = e.parameter.action; const callback = e.parameter.callback; let result; if (action === 'cargarPendientes') result = cargarPendientes(); else if (action === 'cargarIntegrantes') result = cargarIntegrantes(); else result = { ok: false, msg: 'Acción desconocida' }; const json = JSON.stringify(result); const output = callback ? ContentService.createTextOutput(callback + '(' + json + ')').setMimeType(ContentService.MimeType.JAVASCRIPT) : ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON); return output; } // ------------------------------------------------------------ // 1. Guardar reunión completa // ------------------------------------------------------------ function guardarReunion(payload) { const ss = SpreadsheetApp.openById(SHEET_ID); const ts = new Date().toISOString(); // --- Hoja: Reuniones --- const shReuniones = getOrCreate(ss, 'Reuniones', ['ID','Fecha','N°Reunión','Hora','Lugar','Timestamp']); const reunionId = 'RN-' + Utilities.formatDate(new Date(), 'America/Santiago', 'yyyyMMdd-HHmmss'); shReuniones.appendRow([reunionId, payload.fecha, payload.numReunion, payload.hora, payload.lugar, ts]); // --- Hoja: Asistencia --- const shAsist = getOrCreate(ss, 'Asistencia', ['ReunionID','Fecha','Grupo','Nombre','Cargo','Estado']); (payload.asistencia || []).forEach(a => shAsist.appendRow([reunionId, payload.fecha, a.grupo, a.nombre, a.cargo, a.estado])); // --- Hoja: PlanesAccion --- const shPlanes = getOrCreate(ss, 'PlanesAccion', ['ReunionID','Fecha','Descripcion','Responsable','Correo','FechaTermino','Estado','Origen']); (payload.planes || []).forEach(p => shPlanes.appendRow([reunionId, payload.fecha, p.descripcion, p.responsable, p.correo, p.fechaTermino, p.estado, 'reunion'])); (payload.nuevosPlanes || []).forEach(p => shPlanes.appendRow([reunionId, payload.fecha, p.descripcion, p.responsable, p.correo, p.fechaTermino, 'Pendiente', 'nuevo'])); (payload.redefinidos || []).forEach(p => shPlanes.appendRow([reunionId, payload.fecha, p.descripcion + ' [REDEFINIDO: ' + p.motivo + ']', p.responsable, '', p.nuevaFecha, 'Redefinido', 'redefinido'])); // --- Hoja: Inspecciones --- const shInsp = getOrCreate(ss, 'Inspecciones', ['ReunionID','Fecha','Area','Condicion','Riesgo','Responsable','Plazo','Estado']); (payload.inspecciones || []).forEach(i => shInsp.appendRow([reunionId, payload.fecha, i.area, i.condicion, i.riesgo, i.responsable, i.plazo, i.estado])); // --- Hoja: Capacitacion --- const shCap = getOrCreate(ss, 'Capacitacion', ['ReunionID','Fecha','Curso','PersonalObligado','Estado','FechaProgramada','NParticipantes','Obs']); (payload.cursos || []).forEach(c => shCap.appendRow([reunionId, payload.fecha, c.nombre, c.personal, c.estado, c.fechaProgramada, c.nParticipantes, c.obs])); // --- Hoja: ActaAnterior --- const shActa = getOrCreate(ss, 'ActaAnterior', ['ReunionID','Fecha','NumActaAnterior','FechaActaAnterior','Resumen']); shActa.appendRow([reunionId, payload.fecha, payload.numActaAnterior, payload.fechaActaAnterior, payload.resumenActa]); // Enviar acta por correo a todos los integrantes enviarActaPorCorreo(payload, reunionId); return { ok: true, reunionId }; } // ------------------------------------------------------------ // Enviar acta por correo a todos los integrantes // ------------------------------------------------------------ function enviarActaPorCorreo(payload, reunionId) { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName('Integrantes'); if (!sh || sh.getLastRow() < 2) return; // Recopilar destinatarios const rows = sh.getRange(2, 1, sh.getLastRow()-1, 6).getValues(); const destinatarios = rows .map(r => r[2]) .filter(mail => mail && mail.toString().includes('@')); if (!destinatarios.length) return; const fecha = payload.fecha ? Utilities.formatDate(new Date(payload.fecha), 'America/Santiago', 'dd/MM/yyyy') : ''; const asunto = `[CPHS Apprecio] Acta Reunión N°${payload.numReunion || '–'} — ${fecha}`; // Construir HTML del acta const html = buildActaHTML(payload, reunionId, fecha); destinatarios.forEach(mail => { try { GmailApp.sendEmail(mail, asunto, '', { htmlBody: html, name: 'Comité Paritario Apprecio' }); } catch(err) { Logger.log('Error enviando a ' + mail + ': ' + err); } }); } // ------------------------------------------------------------ // Construir HTML del acta // ------------------------------------------------------------ function buildActaHTML(p, reunionId, fecha) { const red = '#fa345e'; // Asistencia const asistencia = (p.asistencia || []); const empresa = asistencia.filter(a => a.grupo === 'empresa'); const trabajadores = asistencia.filter(a => a.grupo === 'trabajadores'); function filaPersona(a) { const color = a.estado === 'Presente' ? '#d4edda' : '#f8f8f8'; const txt = a.estado === 'Presente' ? '#155724' : '#999'; return ` ${a.nombre} ${a.cargo} ${a.estado} `; } // Planes const planes = (p.planes || []).concat(p.nuevosPlanes || []); const filasPlanes = planes.length ? planes.map(pl => ` ${pl.descripcion||'–'} ${pl.responsable||'–'} ${pl.fechaTermino||'–'} ${pl.estado||'–'} `).join('') : 'Sin planes registrados'; // Inspecciones const inspecciones = p.inspecciones || []; const filasInsp = inspecciones.length ? inspecciones.map(i => ` ${i.area||'–'} ${i.condicion||'–'} ${i.riesgo||'–'} ${i.responsable||'–'} ${i.plazo||'–'} `).join('') : 'Sin condiciones registradas'; return `

Acta Reunión CPHS N°${p.numReunion||'–'}

${fecha}  ·  ${p.hora||''}  ·  ${p.lugar||''}

${p.resumenActa ? `

Acta anterior

${p.resumenActa}

` : ''}

1. Asistencia — Empresa

${empresa.map(filaPersona).join('')}
NombreCargoEstado

1. Asistencia — Trabajadores

${trabajadores.map(filaPersona).join('')}
NombreCargoEstado

3. Planes de acción

${filasPlanes}
DescripciónResponsableFecha términoEstado

4. Inspección de seguridad

${filasInsp}
ÁreaCondiciónRiesgoResponsablePlazo
${p.observaciones ? `

Observaciones

${p.observaciones}

` : ''}

ID Reunión: ${reunionId}

Este correo fue generado automáticamente por el sistema CPHS Apprecio.
`; } // ------------------------------------------------------------ // 2. Cargar integrantes desde hoja "Integrantes" // ------------------------------------------------------------ function cargarIntegrantes() { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName('Integrantes'); if (!sh || sh.getLastRow() < 2) return { ok: true, empresa: [], trabajadores: [] }; const rows = sh.getRange(2, 1, sh.getLastRow()-1, 6).getValues(); const empresa = [], trabajadores = []; rows.forEach(r => { const nombre = r[0], cargo = r[1], mail = r[2], grupo = (r[3]||'').toString().trim().toLowerCase(), calidad = r[4], cargoCPHS = r[5]; if (!nombre) return; const obj = { nombre, cargo, mail, calidad, cargoCPHS }; if (grupo === 'empresa') empresa.push(obj); else if (grupo === 'trabajadores') trabajadores.push(obj); }); return { ok: true, empresa, trabajadores }; } // ------------------------------------------------------------ // 3. Cargar datos pendientes para próxima reunión // ------------------------------------------------------------ function cargarPendientes() { const ss = SpreadsheetApp.openById(SHEET_ID); const result = { planes: [], inspecciones: [], ultimaReunion: null }; // Planes no cerrados const shPlanes = ss.getSheetByName('PlanesAccion'); if (shPlanes && shPlanes.getLastRow() > 1) { const rows = shPlanes.getRange(2, 1, shPlanes.getLastRow()-1, 8).getValues(); rows.forEach(r => { if (r[6] !== 'Cerrado' && r[6] !== 'cerrado') { result.planes.push({ reunionId: r[0], fecha: r[1], descripcion: r[2], responsable: r[3], correo: r[4], fechaTermino: r[5], estado: r[6] }); } }); } // Inspecciones no resueltas const shInsp = ss.getSheetByName('Inspecciones'); if (shInsp && shInsp.getLastRow() > 1) { const rows = shInsp.getRange(2, 1, shInsp.getLastRow()-1, 8).getValues(); rows.forEach(r => { if (r[7] !== 'Resuelto') { result.inspecciones.push({ area: r[2], condicion: r[3], riesgo: r[4], responsable: r[5], plazo: r[6], estado: r[7] }); } }); } // Última reunión (para referencia en acta anterior) const shRn = ss.getSheetByName('Reuniones'); if (shRn && shRn.getLastRow() > 1) { const last = shRn.getRange(shRn.getLastRow(), 1, 1, 6).getValues()[0]; result.ultimaReunion = { id: last[0], fecha: last[1], num: last[2] }; } return { ok: true, data: result }; } // ------------------------------------------------------------ // 3. Trigger diario — recordatorios por correo // Configura esto en: Triggers > recordatoriosDiarios > Time-driven > Day timer // ------------------------------------------------------------ function recordatoriosDiarios() { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName('PlanesAccion'); if (!sh || sh.getLastRow() < 2) return; const hoy = new Date(); hoy.setHours(0,0,0,0); const rows = sh.getRange(2, 1, sh.getLastRow()-1, 8).getValues(); rows.forEach((r, i) => { const estado = r[6]; if (estado === 'Cerrado' || estado === 'cerrado') return; const correo = r[4]; if (!correo || !correo.includes('@')) return; const fechaTermino = new Date(r[5]); fechaTermino.setHours(0,0,0,0); const diasRestantes = Math.round((fechaTermino - hoy) / 86400000); if (diasRestantes === 3 || diasRestantes === 1 || diasRestantes === 0) { const descripcion = r[2] || '(sin descripción)'; const responsable = r[3] || 'Responsable'; const alerta = diasRestantes === 0 ? '¡VENCE HOY!' : diasRestantes === 1 ? 'Vence mañana' : 'Vence en 3 días'; const asunto = `[CPHS] ${alerta} — Plan de acción: ${descripcion.substring(0,50)}`; const cuerpo = `Estimado/a ${responsable},\n\n` + `Este es un recordatorio automático del Comité Paritario de Higiene y Seguridad (CPHS).\n\n` + `Tienes un plan de acción próximo a vencer:\n\n` + ` Descripción : ${descripcion}\n` + ` Fecha límite: ${Utilities.formatDate(fechaTermino, 'America/Santiago', 'dd/MM/yyyy')}\n` + ` Estado actual: ${estado}\n\n` + `Por favor actualiza el estado del plan o contacta al coordinador del CPHS.\n\n` + `Saludos,\nComité Paritario de Higiene y Seguridad`; try { GmailApp.sendEmail(correo, asunto, cuerpo); Logger.log('Correo enviado a ' + correo + ' — ' + alerta); // Marcar en sheet que se envió recordatorio sh.getRange(i+2, 8).setNote('Recordatorio enviado: ' + new Date().toLocaleDateString()); } catch(err) { Logger.log('Error enviando a ' + correo + ': ' + err); } } }); } // ------------------------------------------------------------ // Utilidad: obtener o crear hoja con encabezados // ------------------------------------------------------------ function getOrCreate(ss, name, headers) { let sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); sh.appendRow(headers); sh.getRange(1, 1, 1, headers.length).setFontWeight('bold') .setBackground('#1a3a5c').setFontColor('white'); sh.setFrozenRows(1); } return sh; }
